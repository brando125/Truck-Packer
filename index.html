The Published CSV URL pulls the item list. The Sync URL (optional) pulls and saves the layout to a hidden tab in your spreadsheet so anyone with both URLs sees the same packed layout.

How to set up the Sync URL (one-time, ~3 min)
  1. Open your spreadsheet → Extensions → Apps Script.
  2. Delete any default code, paste in the script below, click Save.
  3. Click Deploy → New deployment → gear icon → Web app.
  4. Set Execute as: Me, Who has access: Anyone, click Deploy, authorize.
  5. Copy the Web app URL it shows, paste it above.
const STATE_TAB = 'TruckPackerState';

function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(STATE_TAB);
  const state = sheet ? String(sheet.getRange('A1').getValue() || '') : '';
  const updatedAt = sheet ? String(sheet.getRange('B1').getValue() || '') : '';
  return ContentService.createTextOutput(JSON.stringify({ ok: true, state, updatedAt }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(STATE_TAB);
  if (!sheet) { sheet = ss.insertSheet(STATE_TAB); sheet.hideSheet(); }
  const body = (e && e.postData && e.postData.contents) || '';
  const now = new Date().toISOString();
  sheet.getRange('A1').setValue(body);
  sheet.getRange('B1').setValue(now);
  return ContentService.createTextOutput(JSON.stringify({ ok: true, updatedAt: now }))
    .setMimeType(ContentService.MimeType.JSON);
}
